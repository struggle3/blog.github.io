Spring Boot配置类

2021-03-14

1. 介绍

通过@Configuration,@Import,@Conditional用于自定义配置类与XML配置文件等效，被注解的类内部包含一个或多个@Bean注解方法。可以被AnnotationConfigApplicationContext或者AnnotationConfigWebApplicationContext进行扫描。用于构建bean定义以及初始化Spring容器。

  1. 在配置类中用@Bean注解标注在public方法上将注册组件到容器中，方法作为组件id、返回类型作为组件类型、返回值就是实器中的组件实例、默认为单例。
  2. 配置类本身也是组件。
  3. 使用@Configuration(proxyBeanMethods = false)的true和false指定bean是否为代理方法
    • 最佳实践
      1. 配置类组件之间无依赖关系设置Full(proxyBeanMethods = false)
      2. 配置类组件之间有依赖关系设置Full(proxyBeanMethods = true)
  4. @Import用于导入外部组件配置
  5. @Conditional用于按条件进行配置类的条件限定

2. 总结

@Configuation等价于<Beans>

@Bean 等价于<Bean></Bean>

@ComponentScan等价于<context:component-scan/>

@Component 等价于<Bean></Bean>

3. 实例

  • 配置类
package com.springboot01.config;

import ch.qos.logback.core.db.DBHelper;
import com.springboot01.beans.Person;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Import;


@Import({DBHelper.class})//向容器中导入组件、默认组件的名字就是全类名
@Configuration()//使用@Configuration声明配置类
public class MyConfig {
    @Bean(name = "person")
    public Person person(){

        return new Person("zhangsan");
    };
}


package com.springboot01.beans;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.stereotype.Component;
import org.springframework.validation.annotation.Validated;

import javax.validation.constraints.NotNull;

public class Person {
    public Person(String name) {
        this.name = name;
    }
    private  String name;
    private  String password;
    private  int age;
    private  String address;

    @Override
    public String toString() {
        return "User{" +
                "name='" + name + '\'' +
                ", password='" + password + '\'' +
                ", age=" + age +
                ", address='" + address + '\'' +
                '}';
    }

}
  • 测试方法
package com.springboot01;

import ch.qos.logback.core.db.DBHelper;
import com.springboot01.beans.Person;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.ConfigurableApplicationContext;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.PropertySource;
import org.springframework.context.annotation.PropertySources;

import java.io.IOException;
import java.io.InputStream;
import java.util.Properties;

@SpringBootApplication
public class Springboot01Application {

    public static void main(String[] args) throws IOException
    {
        //1、获取IOC容器
        ConfigurableApplicationContext run = SpringApplication.run(Springboot01Application.class, args);

        //2、获取容器中所有组件
        String[] names = run.getBeanDefinitionNames();
        for (String name : names) {
            System.out.println(name);
        }

        //获取bean
        Person person=run.getBean(Person.class);
        System.out.println(person);

        DBHelper dbHelper = run.getBean(DBHelper.class);
        System.out.println(dbHelper);
    }

}
