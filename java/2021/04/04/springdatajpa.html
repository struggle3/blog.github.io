Specification动态查询实例

2021-04-04

在软件开发中需要对某个实体进行多条件复杂查询时时，由于默认的Repository中的方法和方法名约定查询可能不能满足要求。

可以通过Spring Data JPA中可以通过Specification自定义查询条件实现。

1. 实现思路：

  1. 实现Specification接口（提供泛型：查询的对象类型）
  2. 实现toPredicate方法（构造查询条件）
  3. 通过方法参数中的两个参数（
    • root：查询对象属性
    • CriteriaBuilder：查询条件

2. 实例代码：

import com.dao.UserDao;
import com.domain.User;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Pageable;
import org.springframework.data.domain.Sort;
import org.springframework.data.jpa.domain.Specification;
import org.springframework.test.context.ContextConfiguration;
import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;

import javax.persistence.criteria.*;
import java.util.List;

@RunWith(SpringJUnit4ClassRunner.class) //声明spring提供的单元测试环境
@ContextConfiguration(locations = "classpath:applicationContext.xml")//加载spring容器
public class TestSpe {

    @Autowired
    private UserDao userDao;

    /*
    * 精确查询
    * */
    @Test
    public void Test01(){

        Specification<User> spec = new Specification<User>() {
            @Override
            public Predicate toPredicate(Root<User> root, CriteriaQuery<?> query, CriteriaBuilder cb) {
                //1.获取比较的属性
                Path<Object> p_name = root.get("name");
                //2.构造查询条件
                /**
                 * 第一个参数：比较的属性
                 * 第二个参数：比较的取值
                 */
                Predicate predicate = cb.equal(p_name,"lisi");
                return predicate;
            }
        };

        User user=userDao.findOne(spec);
        System.out.println(user);
    }

    /*
    * 多条件查询
    * */
    @Test
    public void Test02(){

        Specification<User> spec = new Specification<User>() {
            @Override
            public Predicate toPredicate(Root<User> root, CriteriaQuery<?> query, CriteriaBuilder cb) {
                //1.获取比较的属性
                Path<Object> p_name = root.get("name");
                Path<Object> p_username = root.get("userName");
                //2.构造查询条件
                /**
                 * 第一个参数：比较的属性
                 * 第二个参数：比较的取值
                 */
                Predicate p1 = cb.equal(p_name,"zhangsan");
                Predicate p2 = cb.equal(p_username,"zs");
                //3.组合查询条件
                Predicate and =cb.and(p1,p2);

                return and;
            }
        };

        User user=userDao.findOne(spec);
        System.out.println(user);
    }

    /*
    * 模糊查询并排序
    * */
    @Test
    public void Test03() {
        Specification<User> spec = new Specification<User>() {
            @Override
            public Predicate toPredicate(Root<User> root, CriteriaQuery<?> query, CriteriaBuilder cb) {
                //1.获取比较的属性
                Path<Object> password = root.get("Password");
                //2.构造查询条件
                //path.as(类型的字节码对象)
                Predicate like = cb.like(password.as(String.class), "123%");
                return like;
            }
        };
        //添加排序
        Sort sort = new Sort(Sort.Direction.DESC,"id");
        List<User> list = userDao.findAll(spec, sort);
        for (User user : list) {
            System.out.println(user);
        }
    }
    /*
    * 分页并排序
    * */
    @Test
    public void test04() {

        //排序
        Sort sort = new Sort(Sort.Direction.DESC,"id");
        //PageRequest对象是Pageable接口的实现类
        Pageable pageable = new PageRequest(0,4,sort);
        //分页查询
        Page<User> page = userDao.findAll(null, pageable);
        //得到数据集合列表
        for (User user:page.getContent()) {
            System.out.println(user);
        }
        System.out.println(page.getTotalElements());//得到总条数
        System.out.println(page.getTotalPages());//得到总页数
    }
}

注意:结合Java 8的lambda表达式使用Specification接口时，代码可以更加简洁。
