JPA实现增删改查

2021-03-30

1. ORM方式

使用EntityManager相关对象进行数据库操作，EntityManager是一个轻量级，线程不安全并内置一级缓存的对象，通过此类可以完成数据库的常规操作。

import com.domain.User;
import org.junit.Test;

import javax.persistence.EntityManager;
import javax.persistence.EntityManagerFactory;
import javax.persistence.EntityTransaction;
import javax.persistence.Persistence;

public class Test01 {
    /**
     * 保存数据
     */
    @Test
    public void testSave() {
        //1.加载配置文件创建工厂（实体管理器工厂）对象
        EntityManagerFactory factory = Persistence.createEntityManagerFactory("myJpa");
        //2.获取实体管理器
        EntityManager em = factory.createEntityManager();
        //3.事务处理
        //获取事务对象
        EntityTransaction tx = em.getTransaction();
        //开启事务
        tx.begin();
        //4.数据持久化
        User user=new User();
        user.setName("zhangsan");
        user.setUserName("zhangsan");
        //保存，
        em.persist(user);
        //5.提交事务
        tx.commit();
        //6.释放资源
        em.close();
        //关闭工厂
        factory.close();
    }

    /**
     * 查询数据（立即加载）
     */
    @Test
    public void testSelect1() {
        //1.加载配置文件创建工厂（实体管理器工厂）对象
        EntityManagerFactory factory = Persistence.createEntityManagerFactory("myJpa");
        //2.获取实体管理器
        EntityManager em = factory.createEntityManager();
        //3.事务处理
        //获取事务对象
        EntityTransaction tx = em.getTransaction();
        //开启事务
        tx.begin();
        //4.数据持久化
        User user=em.find(User.class,1);
        System.out.println(user);
        //5.提交事务
        tx.commit();
        //6.释放资源
        em.close();
        //关闭工厂
        factory.close();
    }

    /**
     * 查询数据（延迟加载）
     */
    @Test
    public void testSelect2() {
        //1.加载配置文件创建工厂（实体管理器工厂）对象
        EntityManagerFactory factory = Persistence.createEntityManagerFactory("myJpa");
        //2.获取实体管理器
        EntityManager em = factory.createEntityManager();
        //3.事务处理
        //获取事务对象
        EntityTransaction tx = em.getTransaction();
        //开启事务
        tx.begin();
        //4.数据持久化
        User user=em.getReference(User.class,1);
        System.out.println(user);
        //5.提交事务
        tx.commit();
        //6.释放资源
        em.close();
        //关闭工厂
        factory.close();
    }

    /**
     * 删除数据
     */
    @Test
    public void testDelete() {
        //1.加载配置文件创建工厂（实体管理器工厂）对象
        EntityManagerFactory factory = Persistence.createEntityManagerFactory("myJpa");
        //2.获取实体管理器
        EntityManager em = factory.createEntityManager();
        //3.事务处理
        //获取事务对象
        EntityTransaction tx = em.getTransaction();
        //开启事务
        tx.begin();
        //4.数据持久化
        User user=em.getReference(User.class,1);
        em.remove(user);
        System.out.println(user);
        //5.提交事务
        tx.commit();
        //6.释放资源
        em.close();
        //关闭工厂
        factory.close();
    }

    /**
     * 更新数据
     */
    @Test
    public void testUpdate() {
        //1.加载配置文件创建工厂（实体管理器工厂）对象
        EntityManagerFactory factory = Persistence.createEntityManagerFactory("myJpa");
        //2.获取实体管理器
        EntityManager em = factory.createEntityManager();
        //3.事务处理
        //获取事务对象
        EntityTransaction tx = em.getTransaction();
        //开启事务
        tx.begin();
        //4.数据持久化
        User user=em.getReference(User.class,2);
        System.out.println(user);
        user.setName("张三");
        em.clear();
        em.merge(user);
        System.out.println(user);
        //5.提交事务
        tx.commit();
        //6.释放资源
        em.close();
        //关闭工厂
        factory.close();
    }

}

2. JPQL方式

基于首次在EJB2.0中引入的EJB查询语言(EJB QL),Java持久化查询语言(JPQL)是一种可移植的查询语言，旨在以面向对象表达式语言的表达式，将SQL语法和简单查询语义绑定在一起。

使用这种语言编写的查询是可移植的，可以被编译成所有主流数据库服务器上的SQL，其特征与原生SQL语句类似并且完全面向对象。

import org.junit.Test;

import javax.persistence.*;
import java.util.List;

public class Test02 {

    /**
     * 查询所有
     */
    @Test
    public void selectAll() {
        //1.加载配置文件创建工厂（实体管理器工厂）对象
        EntityManagerFactory factory = Persistence.createEntityManagerFactory("myJpa");
        //2.获取实体管理器
        EntityManager em = factory.createEntityManager();
        //3.事务处理
        //获取事务对象
        EntityTransaction tx = em.getTransaction();
        //开启事务
        tx.begin();
        //4.数据持久化
        String jpql = "from User";
        Query query = em.createQuery(jpql);
        //查询并返回结果
        List list = query.getResultList();
        for (Object object : list) {
            System.out.println(object);
        }
        //5.提交事务
        tx.commit();
        //6.释放资源
        em.close();
        //关闭工厂
        factory.close();
    }

    /**
